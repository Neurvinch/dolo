id: dataflow_multi_agent_orchestration

tasks:
  # STEP 1: FETCH DATA FROM 5 SOURCES

    url: "https://api.example.com/data"

      Authorization: "Bearer {{ secret('api_token') }}"
  
  - id: fetch_database_data
    type: io.kestra.plugin.core.log.Log
    message: "SELECT * FROM metrics WHERE timestamp > NOW() - INTERVAL 5 MINUTES"
  
  - id: fetch_csv_data
    type: io.kestra.plugin.core.http.Download
    uri: "https://example.com/data.csv"
  
  - id: fetch_webhook_data
    type: io.kestra.plugin.core.http.Request
    url: "https://webhook-collector.com/latest"
  
  - id: fetch_third_party_api
    type: io.kestra.plugin.core.http.Request
    url: "https://stripe.com/v1/charges"

  # STEP 2: AI AGENT SUMMARIZATION (â­ WAKANDA AWARD)
  
  - id: summarize_api_data
    type: io.kestra.plugin.ai.agent.AIAgent
    systemMessage: |
      You are a data analyst specializing in API response analysis.

      Format output as JSON: {"summary": "...", "key_metrics": {...}, "alerts": [...]}
    prompt: |
      Summarize this API response data:
      {{ outputs.fetch_api_data.body }}
  
  - id: summarize_database_data
    type: io.kestra.plugin.ai.agent.AIAgent

      You are a database analyst.

      Analyze this database data:

  - id: summarize_csv_data
    type: io.kestra.plugin.ai.agent.AIAgent
    systemMessage: |
      You are a CSV data expert.
      Summarize CSV contents with statistical insights.
    prompt: |
      Analyze this CSV:
      {{ outputs.fetch_csv_data.body }}
  
  - id: summarize_webhook_data
    type: io.kestra.plugin.ai.agent.AIAgent
    systemMessage: |
      You are a webhook event analyzer.
      Summarize recent events and patterns.
    prompt: |
      Summarize these webhook events:
      {{ outputs.fetch_webhook_data.body }}

    type: io.kestra.plugin.ai.agent.AIAgent
    systemMessage: |
      You are a third-party API analyst.
      Extract valuable insights from external service data.

      Analyze this third-party data:
      {{ outputs.fetch_third_party_api.body }}

  # STEP 3: SYNTHESIS & DECISION AGENT (â­ WAKANDA AWARD - DECISIONS)

      {
        "overall_status": "normal|warning|critical",
        "confidence": 0.95,
        "decision": "description of decision",
        "recommended_action": "specific action to take",
        "affected_systems": ["system1", "system2"],
        "priority": "high|medium|low"
      }

      Here are summaries from 5 different data sources:
      
      API Data Summary: {{ outputs.summarize_api_data.output }}
      Database Summary: {{ outputs.summarize_database_data.output }}
      CSV Summary: {{ outputs.summarize_csv_data.output }}
      Webhook Summary: {{ outputs.summarize_webhook_data.output }}
